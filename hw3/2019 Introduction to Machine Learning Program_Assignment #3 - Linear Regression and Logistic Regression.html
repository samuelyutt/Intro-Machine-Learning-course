def innerprdt(a, b):
    prdt = 0;
    for i in range( len(a) ):
        prdt += (a[i] * b[i])
    return prdt

def sigmoid(wt, x):
    try:
        num = math.exp( -innerprdt(wt, x) )
    except OverflowError:
        num = float('inf')
    return 1/(1+num)
