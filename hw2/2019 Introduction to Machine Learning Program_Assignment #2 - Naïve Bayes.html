        def entropy(data):
    v_count = len(data)
    t_count = 0
    if v_count == 0:
        return 0
    for i in range(0, v_count):
        if int(data[i][ len(data[i])-1 ]) == 1:
            t_count += 1
    f_count = v_count - t_count
    p_t = t_count / v_count
    p_f = f_count / v_count
    if (p_t) == 0:
        t_entropy = 0
    else:
        t_entropy = p_t * math.log(p_t)

    if (p_f) == 0:
        f_entropy = 0
    else:
        f_entropy = p_f * math.log(p_f) 
    entropy = -( t_entropy + f_entropy )
    return entropy

        def gini_index(data):
    v_count = len(data)
    t_count = 0
    if v_count == 0:
        return 0
    for i in range(0, v_count):
        if int(data[i][ len(data[i])-1 ]) == 1:
            t_count += 1
    f_count = v_count - t_count
    p_t = t_count / v_count
    p_f = f_count / v_count
    if (p_t) == 0:
        t_gini = 0
    else:
        t_gini = p_t * p_t

    if (p_f) == 0:
        f_gini = 0
    else:
        f_gini = p_f * p_f
    gini = 1 - ( t_gini + f_gini )   
    return gini

        • K-ford cross-validation chooses 1/k data to be validation data and left the other to be training data. After one train finished, it will choose another 1/k data and so on.
